157  void Font::Draw(const D3DXVECTOR2* _pDrawPos, LPCTSTR _pStr)
158  {
159  m_pGraphicsDevice->GetDeviceContext()->VSSetShader(m_pVertexShader, NULL, 0);
160  m_pGraphicsDevice->GetDeviceContext()->PSSetShader(m_pPixelShader, NULL, 0);
161 
162  m_pGraphicsDevice->GetDeviceContext()->VSSetConstantBuffers(0, 1, &m_pConstantBuffer);
163  m_pGraphicsDevice->GetDeviceContext()->PSSetConstantBuffers(0, 1, &m_pConstantBuffer);
164 
165  m_pGraphicsDevice->GetDeviceContext()->IASetInputLayout(m_pVertexLayout);
166  m_pGraphicsDevice->GetDeviceContext()->IASetPrimitiveTopology(D3D11_PRIMITIVE_TOPOLOGY_TRIANGLESTRIP);
167 
168  m_pGraphicsDevice->GetDeviceContext()->PSSetSamplers(0, 1, &m_pSamplerState);
169  m_pGraphicsDevice->GetDeviceContext()->PSSetShaderResources(0, 1, &m_pShaderResource);
170 
171  m_pGraphicsDevice->GetDeviceContext()->OMSetBlendState(m_pBlendState, NULL, 0xffffffff);
172 
173  UINT Stride = sizeof(FONT_VERTEX);
174  UINT Offset = 0;
175  m_pGraphicsDevice->GetDeviceContext()->IASetVertexBuffers(0, 1, &m_pVertexBuffer, &Stride, &Offset);
176 
177 
178  D3DXMATRIX MatWorld, MatTranslate, MatRotate;
179  D3DXMatrixIdentity(&MatWorld);
180  D3DXMatrixScaling(&MatWorld, 1.0f, 1.0f, 1.0f);
181  D3DXMatrixRotationZ(&MatRotate, 0.0f);
182  D3DXMatrixMultiply(&MatWorld, &MatWorld, &MatRotate);
183  D3DXMatrixTranslation(&MatTranslate, _pDrawPos->x, _pDrawPos->y, 0.0f);
184  D3DXMatrixMultiply(&MatWorld, &MatWorld, &MatTranslate);
185 
186  for (int i = 0; i < static_cast<int>(strlen(_pStr)); i++)
187  {
188  D3D11_MAPPED_SUBRESOURCE MappedResource;
189  CONSTANT_BUFFER ConstantBuffer;
190  m_pGraphicsDevice->GetDeviceContext()->Map(m_pConstantBuffer, 0, D3D11_MAP_WRITE_DISCARD, 0, &MappedResource);
191 
192  ConstantBuffer.MatWorld = MatWorld;
193  D3DXMatrixTranspose(&ConstantBuffer.MatWorld, &ConstantBuffer.MatWorld);
194 
195  ConstantBuffer.WindowSize.x = m_WindowWidth;
196  ConstantBuffer.WindowSize.y = m_WindowHeight;
197 
198  ConstantBuffer.TexelOffset.x = static_cast<float>(_pStr[i] - m_SpaceAsciiCode) * m_FontTu;
199  ConstantBuffer.TexelOffset.y = 0.0f;
200  ConstantBuffer.TexelOffset.z = 0.0f;
201  ConstantBuffer.TexelOffset.w = 0.0f;
202 
203  ConstantBuffer.PosOffset.x = i * m_FontSize.x;
204  ConstantBuffer.PosOffset.y = 0;
205  ConstantBuffer.PosOffset.z = 0;
206  ConstantBuffer.PosOffset.w = 0;
207 
208  memcpy_s(
209  MappedResource.pData,
210  MappedResource.RowPitch,
211  reinterpret_cast<void*>(&ConstantBuffer),
212  sizeof(ConstantBuffer));
213 
214  m_pGraphicsDevice->GetDeviceContext()->Unmap(m_pConstantBuffer, 0);
215 
216  m_pGraphicsDevice->GetDeviceContext()->Draw(VERTEX_NUM, 0);
217  }
