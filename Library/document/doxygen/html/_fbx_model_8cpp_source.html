174  bool FbxModel::CreateIndexBuffer()
175  {
176  m_ppIndexBuffer = new ID3D11Buffer*[m_MeshData.size()];
177 
178  for (unsigned int MeshIndex = 0; MeshIndex < m_MeshData.size(); MeshIndex++)
179  {
180  int PolygonVertexNum = m_MeshData[MeshIndex].pVertexData->PolygonVertexNum; // メッシュの総頂点数.
181 
182  // インデックスバッファの設定.
183  D3D11_BUFFER_DESC BufferDesc;
184  ZeroMemory(&BufferDesc, sizeof(D3D11_BUFFER_DESC));
185  BufferDesc.ByteWidth = sizeof(int) * PolygonVertexNum;
186  BufferDesc.Usage = D3D11_USAGE_DEFAULT;
187  BufferDesc.BindFlags = D3D11_BIND_INDEX_BUFFER;
188  BufferDesc.CPUAccessFlags = 0;
189  BufferDesc.MiscFlags = 0;
190  BufferDesc.StructureByteStride = 0;
191 
192  // インデックスバッファに格納するデータの設定.
193  D3D11_SUBRESOURCE_DATA ResourceData;
194  ZeroMemory(&ResourceData, sizeof(ResourceData));
195  ResourceData.pSysMem = m_MeshData[MeshIndex].pVertexData->pIndexAry;
196  ResourceData.SysMemPitch = 0;
197  ResourceData.SysMemSlicePitch = 0;
198 
199  // インデックスバッファの生成.
200  if (FAILED(m_pGraphicsDevice->GetDevice()->CreateBuffer(
201  &BufferDesc,
202  &ResourceData,
203  &m_ppIndexBuffer[MeshIndex])))
204  {
205  OutputErrorLog("インデックスバッファの作成に失敗しました");
206 
207  for (unsigned int i = 0; i < MeshIndex; i++)
208  {
209  SafeRelease(m_ppIndexBuffer[i]);
210  }
211  SafeDeleteArray(m_ppIndexBuffer);
212 
213  return false;
214  }
215  }
216 
217  return true;

220  bool FbxModel::CreateVertexBuffer()
221  {
222  m_ppVertexBuffer = new ID3D11Buffer*[m_MeshData.size()];
223  m_ppVertexData = new FBXMODEL_VERTEX*[m_MeshData.size()];
224 
225  for (unsigned int MeshIndex = 0; MeshIndex < m_MeshData.size(); MeshIndex++)
226  {
227  MESH_DATA MeshData = m_MeshData[MeshIndex]; // 現在のメッシュデータ.
228 
229  int ControlPositionNum = MeshData.pVertexData->ControlPositionNum; // メッシュのインデックスバッファが指す総頂点数.
230  D3DXVECTOR3* pVertex = MeshData.pVertexData->pVertex; // メッシュが持つ頂点座標の配列.
231  D3DXVECTOR3* pNormal = MeshData.pNormalData->pNormalVec; // メッシュが持つ法線ベクトルの配列.
233  D3DXVECTOR2* pUVVertex = MeshData.pTextureData->pTextureUVData[0].pTextureUV; // メッシュが持つテクスチャ座標の配列.
234 
235 
236  // メッシュに格納されているデータを設定.
237  m_ppVertexData[MeshIndex] = new FBXMODEL_VERTEX[ControlPositionNum];
238  for (int i = 0; i < ControlPositionNum; i++)
239  {
240  m_ppVertexData[MeshIndex][i].Pos = pVertex[i];
241  m_ppVertexData[MeshIndex][i].Normal = pNormal[i];
242  m_ppVertexData[MeshIndex][i].Texel = pUVVertex[i];
243  }
244 
245 
246  // 頂点バッファの設定.
247  D3D11_BUFFER_DESC BufferDesc;
248  ZeroMemory(&BufferDesc, sizeof(D3D11_BUFFER_DESC));
249  BufferDesc.ByteWidth = sizeof(FBXMODEL_VERTEX) * ControlPositionNum;
250  BufferDesc.Usage = D3D11_USAGE_DEFAULT;
251  BufferDesc.BindFlags = D3D11_BIND_VERTEX_BUFFER;
252  BufferDesc.CPUAccessFlags = 0;
253  BufferDesc.MiscFlags = 0;
254 
255  // 頂点バッファに格納するデータの設定.
256  D3D11_SUBRESOURCE_DATA ResourceData;
257  ZeroMemory(&ResourceData, sizeof(ResourceData));
258  ResourceData.pSysMem = m_ppVertexData[MeshIndex];
259  ResourceData.SysMemPitch = 0;
260  ResourceData.SysMemSlicePitch = 0;
261 
262  // 頂点バッファの生成.
263  if (FAILED(m_pGraphicsDevice->GetDevice()->CreateBuffer(
264  &BufferDesc,
265  &ResourceData,
266  &m_ppVertexBuffer[MeshIndex])))
267  {
268  OutputErrorLog("頂点バッファの作成に失敗しました");
269 
270  for (unsigned int i = 0; i < MeshIndex; i++)
271  {
272  SafeRelease(m_ppVertexBuffer[i]);
273  }

276  return false;
