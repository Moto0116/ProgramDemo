219  bool GraphicsDevice::CreateDisplay()
220  {
221  // グラフィックインストラクチャの取得.
222  if (FAILED(m_pDevice->QueryInterface(__uuidof(IDXGIDevice1), reinterpret_cast<void**>(&m_pDXGI))))
223  {
224  OutputErrorLog("グラフィックインストラクチャの取得に失敗しました");
225  return false;
226  }
227 
228  // グラフィックインストラクチャアダプタの取得.
229  if (FAILED(m_pDXGI->GetAdapter(&m_pAdapter)))
230  {
231  OutputErrorLog("グラフィックインストラクチャアダプタの取得に失敗しました");
232  ReleaseDisplay();
233  return false;
234  }
235 
236  // グラフィックインストラクチャファクトリの取得.
237  if (FAILED(m_pAdapter->GetParent(__uuidof(IDXGIFactory), reinterpret_cast<void**>(&m_pDXGIFactory))))

240  ReleaseDisplay();
241  return false;

243 
244  // DXGIにウィンドウメッセージキューの監視を設定.
245  if (FAILED(m_pDXGIFactory->MakeWindowAssociation(m_hWnd, 0)))
246  {

248  ReleaseDisplay();
249  return false;
250  }
251 
252 
253  // スワップチェインの設定.
254  DXGI_SWAP_CHAIN_DESC SwapChainDesc;
255  SwapChainDesc.BufferDesc.Width = m_WindowRect.right - m_WindowRect.left;
256  SwapChainDesc.BufferDesc.Height = m_WindowRect.bottom - m_WindowRect.top;

258  SwapChainDesc.BufferDesc.RefreshRate.Denominator = 1;
259  SwapChainDesc.BufferDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
260  SwapChainDesc.BufferDesc.ScanlineOrdering = DXGI_MODE_SCANLINE_ORDER_UNSPECIFIED;
261  SwapChainDesc.BufferDesc.Scaling = DXGI_MODE_SCALING_UNSPECIFIED;
262  SwapChainDesc.SampleDesc.Count = 1;
263  SwapChainDesc.SampleDesc.Quality = 0;
264  SwapChainDesc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
265  SwapChainDesc.BufferCount = 1;
266  SwapChainDesc.OutputWindow = m_hWnd;
267  SwapChainDesc.Windowed = TRUE;
268  SwapChainDesc.SwapEffect = DXGI_SWAP_EFFECT_DISCARD;
269  SwapChainDesc.Flags = DXGI_SWAP_CHAIN_FLAG_ALLOW_MODE_SWITCH;
270 
271  // スワップチェインの生成.
272  if (FAILED(m_pDXGIFactory->CreateSwapChain(m_pDevice, &SwapChainDesc, &m_pDXGISwapChain)))
273  {
274  OutputErrorLog("スワップチェインの作成に失敗しました");
275  ReleaseDisplay();
276  return false;
277  }

280  if (FAILED(m_pDXGISwapChain->GetBuffer(0, __uuidof(ID3D11Texture2D), reinterpret_cast<void**>(&m_pBackBuffer))))

308  DepthDesc.BindFlags = D3D11_BIND_DEPTH_STENCIL;
309  DepthDesc.CPUAccessFlags = 0;
310  DepthDesc.MiscFlags = 0;
311 
312  // 深度ステンシルバッファの生成.
313  if (FAILED(m_pDevice->CreateTexture2D(&DepthDesc, NULL, &m_pDepthStencilBuffer)))
314  {

319

321  if (FAILED(m_pDevice->CreateDepthStencilView(
322  m_pDepthStencilBuffer,
323  NULL,
324  &m_pDepthStencilView[BACKBUFFER_TARGET])))

327  ReleaseDisplay();
328  return false;
329  }
330 
331  m_pDeviceContext->OMSetRenderTargets(
332  1,
333  &m_pRenderTargetView[BACKBUFFER_TARGET],
334  m_pDepthStencilView[BACKBUFFER_TARGET]); // 描画先に設定.
335 
336

338  ZeroMemory(&m_ViewPort, sizeof(m_ViewPort));
339  m_ViewPort[BACKBUFFER_TARGET].TopLeftX = 0;
340  m_ViewPort[BACKBUFFER_TARGET].TopLeftY = 0;
341  m_ViewPort[BACKBUFFER_TARGET].Width = static_cast<float>(m_WindowRect.right - m_WindowRect.left);
342  m_ViewPort[BACKBUFFER_TARGET].Height = static_cast<float>(m_WindowRect.bottom - m_WindowRect.top);
343  m_ViewPort[BACKBUFFER_TARGET].MinDepth = 0.0f;
344  m_ViewPort[BACKBUFFER_TARGET].MaxDepth = 1.0f;
345  m_pDeviceContext->RSSetViewports(1, &m_ViewPort[BACKBUFFER_TARGET]);
346

358  RasterizerDesc.MultisampleEnable = FALSE;
359  RasterizerDesc.AntialiasedLineEnable = FALSE;
360 
361  // ラスタライザステートの生成.
362  if (FAILED(m_pDevice->CreateRasterizerState(&RasterizerDesc, &m_pRasterizerState)))
363  {
364  OutputErrorLog("ラスタライザステートの作成に失敗しました");
365  ReleaseDisplay();
366  return false;
367  }
368 
369  m_pDeviceContext->RSSetState(m_pRasterizerState); // ラスタライザステートをセット.
370 
371  return true;
