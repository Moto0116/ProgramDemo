208  bool Vertex2D::WriteConstantBuffer(const D3DXVECTOR2* _pDrawPos, const D3DXVECTOR2* _pScale, float _angle)
209  {
210  D3DXMATRIX MatWorld, MatTranslate, MatRotate;
211  D3DXMatrixIdentity(&MatWorld);
212  D3DXMatrixScaling(&MatWorld, _pScale->x, _pScale->y, 1.0f);
213  D3DXMatrixRotationZ(&MatRotate, _angle);
214  D3DXMatrixMultiply(&MatWorld, &MatWorld, &MatRotate);
215  D3DXMatrixTranslation(&MatTranslate, _pDrawPos->x, _pDrawPos->y, 0.0f);
216  D3DXMatrixMultiply(&MatWorld, &MatWorld, &MatTranslate);
217 
218  D3D11_MAPPED_SUBRESOURCE MappedResource;
219  if (SUCCEEDED(m_pGraphicsDevice->GetDeviceContext()->Map(m_pConstantBuffer, 0, D3D11_MAP_WRITE_DISCARD, 0, &MappedResource)))
220  {
221  CONSTANT_BUFFER ConstantBuffer;
222  ConstantBuffer.MatWorld = MatWorld;
223  D3DXMatrixTranspose(&ConstantBuffer.MatWorld, &ConstantBuffer.MatWorld);
224 
225  ConstantBuffer.WindowSize.x = m_WindowWidth;
226  ConstantBuffer.WindowSize.y = m_WindowHeight;
227 
228  memcpy_s(
229  MappedResource.pData,
230  MappedResource.RowPitch,
231  reinterpret_cast<void*>(&ConstantBuffer),
232  sizeof(ConstantBuffer));
233  m_pGraphicsDevice->GetDeviceContext()->Unmap(m_pConstantBuffer, 0);
234 
235  return true;
236  }
