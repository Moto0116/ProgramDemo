127  FILE* pAnimationFile = NULL;
128  std::vector<float> AnimationData;
129  long FileSize = 0;
130  char* pBuffer = NULL;
131  char* pContext = NULL;
132  char* pDataStr = NULL;
133 
134  fopen_s(&pAnimationFile, _pAnimationPath, "r");
135 
136  // ファイルサイズの取得.
137  fseek(pAnimationFile, 0, SEEK_END);
138  FileSize = ftell(pAnimationFile) + 1;
139  fseek(pAnimationFile, 0, SEEK_SET);
140 
141  // ファイルのデータを格納するバッファ.
142  pBuffer = new char[FileSize];
143  ZeroMemory(pBuffer, FileSize);
144 
145  // ファイルの読み込み.
146  fread(pBuffer, FileSize, 1, pAnimationFile);
147  pBuffer[FileSize - 1] = '\0';
148 
149  // ファイル内のデータを見つけて取得.
150  pBuffer = strstr(pBuffer, "{");
151  strcpy_s(pBuffer, FileSize, strtok_s(pBuffer, "{}", &pContext));
152  pDataStr = strtok_s(pBuffer, "{},\n", &pContext);
153 
154  while (1)
155  {
156  if (pDataStr == NULL)
157  {
158  break; // データ全てを取り出すまでループ
159  }
160  else
161  {
162  AnimationData.push_back(static_cast<float>(atof(pDataStr)));
163  }
164 
165  pDataStr = strtok_s(NULL, "{},\n", &pContext);
166  }
167 
168  fclose(pAnimationFile);
169 
170 
171  // 取り出したデータを追加する.
172  for (unsigned int i = 0; i < AnimationData.size() / 5; i++)
173  {
174  m_pFrame.push_back(new ANIMATION_FRAME);
175  m_pFrame[i]->FrameID = static_cast<int>(AnimationData[i * 5]);
176  m_pFrame[i]->MinTu = AnimationData[i * 5 + 1];
177  m_pFrame[i]->MinTv = AnimationData[i * 5 + 2];

179  m_pFrame[i]->MaxTv = AnimationData[i * 5 + 4];
180  }
