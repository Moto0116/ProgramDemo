314  bool FbxLoader::LoadMeshTextureData(FbxMesh* _pMesh, FbxModel::MESH_DATA* _pMeshData)
315  {
316  std::vector<fbxsdk::FbxString> UvSetName;
317  int UVSetCount = _pMesh->GetElementUVCount();
318 
319  FbxModel::TEXTURE_DATA* pTextureData = new FbxModel::TEXTURE_DATA; // UVセットの数確保.
320  pTextureData->pTextureUVData = new FbxModel::TEXTURE_UV_DATA[UVSetCount];
321  pTextureData->TextureUVCount = UVSetCount;
322  D3DXVECTOR2** ppTextureUV = new D3DXVECTOR2*[UVSetCount];
323  for (int i = 0; i < UVSetCount; i++)
324  {
325  ppTextureUV[i] = new D3DXVECTOR2[_pMeshData->pVertexData->PolygonVertexNum];
326  }
327 
328 
329  for (int i = 0; i < UVSetCount; i++)
330  {
331  fbxsdk::FbxGeometryElementUV* UVSet = _pMesh->GetElementUV(i); // UVセットの取得.
332  FbxGeometryElement::EMappingMode MappingMode = UVSet->GetMappingMode(); // マッピングモードの取得.
333  FbxGeometryElement::EReferenceMode ReferenceMode = UVSet->GetReferenceMode(); // リファレンスモード取得.
334 
335  switch (MappingMode)
336  {
337  case FbxGeometryElement::eByPolygonVertex:
338 
339  switch (ReferenceMode)
340  {
341  case FbxGeometryElement::eDirect:
342  {
343  for (int n = 0; n < _pMeshData->pVertexData->PolygonVertexNum; n++)
344  {
345  ppTextureUV[i][n].x = static_cast<float>(UVSet->GetDirectArray().GetAt(n)[0]);
346  ppTextureUV[i][n].y = 1.0f - static_cast<float>(UVSet->GetDirectArray().GetAt(n)[1]);
347  }
348 
349  UvSetName.push_back(UVSet->GetName());
350  }
351  break;
352  case FbxGeometryElement::eIndexToDirect:
353  {
354  FbxLayerElementArrayTemplate<int>* UVIndex = &UVSet->GetIndexArray();
355  for (int n = 0; n < _pMeshData->pVertexData->PolygonVertexNum; n++)
356  {
357  int Index = UVIndex->GetAt(n);
358  ppTextureUV[i][n].x = static_cast<float>(UVSet->GetDirectArray().GetAt(Index)[0]);
359  ppTextureUV[i][n].y = 1.0f - static_cast<float>(UVSet->GetDirectArray().GetAt(Index)[1]);
360  }
361 
362  UvSetName.push_back(UVSet->GetName());
363  }
364  break;
365  default:
366  return false;
367  break;
368  }
369 
370  break;
371  default:
372  return false;
373  break;
374  }
375 
376  pTextureData->pTextureUVData[i].pTextureUV = ppTextureUV[i];
377  pTextureData->pTextureUVData[i].pUVSetName = UVSet->GetName();
378  }
379 
380  delete[] ppTextureUV;
381 
382  _pMeshData->pTextureData = pTextureData;
383 
384  return true;

387  bool FbxLoader::LoadMeshMaterialData(FbxMesh* _pMesh, FbxModel::MESH_DATA* _pMeshData)
388  {
389  std::vector<FbxModel::MATERIAL> pMaterial;
390  int TextureFileCount = 0;
391  std::vector<const char*> TextureFileName;
392  std::vector<fbxsdk::FbxString> TextureUvSetName;
393 
394 
395  fbxsdk::FbxNode* Node = _pMesh->GetNode();
396  int MaterialCount = Node->GetMaterialCount();
397 
398  FbxModel::MATERIAL_DATA* pMaterialData = new FbxModel::MATERIAL_DATA[MaterialCount];
399 
400 
401  for (int i = 0; i < MaterialCount; i++)
402  {
403  fbxsdk::FbxSurfaceMaterial* Material = Node->GetMaterial(i); // マテリアルの取得.
404 
405  if (Material->GetClassId().Is(fbxsdk::FbxSurfaceLambert::ClassId))
406  {
407  // Lambertにダウンキャスト.
408  fbxsdk::FbxSurfaceLambert* lambert = reinterpret_cast<fbxsdk::FbxSurfaceLambert*>(Material);
409 
410  FbxModel::MATERIAL MaterialData;
411  ZeroMemory(&MaterialData, sizeof(MaterialData));
412 
413  // アンビエントの取得.
414  MaterialData.Ambient.r = (float)lambert->Ambient.Get().mData[0] * (float)lambert->AmbientFactor.Get();
415  MaterialData.Ambient.g = (float)lambert->Ambient.Get().mData[1] * (float)lambert->AmbientFactor.Get();
416  MaterialData.Ambient.b = (float)lambert->Ambient.Get().mData[2] * (float)lambert->AmbientFactor.Get();
417  GetTextureName(lambert, fbxsdk::FbxSurfaceMaterial::sAmbient, &TextureFileName, &TextureUvSetName, &TextureFileCount);
418 
419  MaterialData.Ambient = m_DefaultAmbient;
420 
421 
422  // ディフューズの取得.
423  MaterialData.Diffuse.r = (float)lambert->Diffuse.Get().mData[0] * (float)lambert->DiffuseFactor.Get();
424  MaterialData.Diffuse.g = (float)lambert->Diffuse.Get().mData[1] * (float)lambert->DiffuseFactor.Get();
425  MaterialData.Diffuse.b = (float)lambert->Diffuse.Get().mData[2] * (float)lambert->DiffuseFactor.Get();
426  GetTextureName(lambert, fbxsdk::FbxSurfaceMaterial::sDiffuse, &TextureFileName, &TextureUvSetName, &TextureFileCount);
427 
428  // エミッシブの取得.
429  MaterialData.Emissive.r = (float)lambert->Emissive.Get().mData[0] * (float)lambert->EmissiveFactor.Get();
430  MaterialData.Emissive.g = (float)lambert->Emissive.Get().mData[1] * (float)lambert->EmissiveFactor.Get();
431  MaterialData.Emissive.b = (float)lambert->Emissive.Get().mData[2] * (float)lambert->EmissiveFactor.Get();
432  GetTextureName(lambert, fbxsdk::FbxSurfaceMaterial::sEmissive, &TextureFileName, &TextureUvSetName, &TextureFileCount);
433 
434  // 透過度の取得.
435  MaterialData.Ambient.a = (float)lambert->TransparentColor.Get().mData[0];
436  MaterialData.Diffuse.a = (float)lambert->TransparentColor.Get().mData[1];
437  MaterialData.Emissive.a = (float)lambert->TransparentColor.Get().mData[2];
438  GetTextureName(lambert, fbxsdk::FbxSurfaceMaterial::sTransparentColor, &TextureFileName, &TextureUvSetName, &TextureFileCount);
439 
440 
441  // 法線マップの取得.
442  GetTextureName(lambert, fbxsdk::FbxSurfaceMaterial::sNormalMap, &TextureFileName, &TextureUvSetName, &TextureFileCount);
443 
444  pMaterial.push_back(MaterialData);
445  }
446  else
447  {
448  return false;
449  }
450 
451  pMaterialData[i].pMaterial = pMaterial[i];
452  pMaterialData[i].TextureCount = TextureFileCount;
453  pMaterialData[i].pTextureName = new LPCTSTR[TextureFileCount];
454  pMaterialData[i].pTextureUVSetName = new LPCTSTR[TextureFileCount];
455  pMaterialData[i].pTextureView = new ID3D11ShaderResourceView*[TextureFileCount];
456  for (int n = 0; n < TextureFileCount; n++)
457  {
458  pMaterialData[i].pTextureName[n] = TextureFileName[n];
459  pMaterialData[i].pTextureUVSetName[n] = TextureUvSetName[n];
460 
461  // テクスチャ読み込みの設定.
462  D3DX11_IMAGE_LOAD_INFO LoadInfo;
463  ZeroMemory(&LoadInfo, sizeof(D3DX11_IMAGE_LOAD_INFO));
464  LoadInfo.Width = D3DX11_DEFAULT;
465  LoadInfo.Height = D3DX11_DEFAULT;
466  LoadInfo.Depth = D3DX11_DEFAULT;
467  LoadInfo.FirstMipLevel = D3DX11_DEFAULT;
468  LoadInfo.MipLevels = 1;
469  LoadInfo.Usage = D3D11_USAGE_DEFAULT;
470  LoadInfo.BindFlags = D3D11_BIND_SHADER_RESOURCE;
471  LoadInfo.CpuAccessFlags = 0;
472  LoadInfo.MiscFlags = 0;
473  LoadInfo.Format = DXGI_FORMAT_FROM_FILE;
474  LoadInfo.Filter = D3DX11_FILTER_POINT;
475  LoadInfo.MipFilter = D3DX11_FILTER_POINT;
476  LoadInfo.pSrcInfo = NULL;
477 
478  ID3D11ShaderResourceView* pResourceView = NULL;
479  if (FAILED(D3DX11CreateShaderResourceViewFromFile(
480  m_pGraphicsDevice->GetDevice(),
481  TextureFileName[n],
482  &LoadInfo,
483  NULL,
484  &pResourceView,
485  NULL)))
486  {
487  OutputErrorLog("テクスチャの読み込みに失敗しました");
488  return false;
489  }
490 
491  pMaterialData[i].pTextureView[n] = pResourceView;
492  }
493  }
494 
495  _pMeshData->pMaterialData = pMaterialData;
496 
497  return true;
498  }
